全局调用

````typescript jsx
import { useGlobalModal, GlobalModalManager } from '@/components/ui/cyber-modal/global-modal';

// 在组件内使用 Hook
function ComponentA() {
    const modal = useGlobalModal();

    const handleClick = () => {
        modal.open({
            title: '全局弹窗 A',
            content: '通过 Hook 打开',
            theme: 'neon',
            onConfirm: () => {
                console.log('确认');
                modal.closeAll();
            }
        });
    };

    return <button onClick={handleClick}>打开全局弹窗</button>;
}

// 在任意地方静态调用
function someUtilityFunction() {
    const modalId = GlobalModalManager.open({
        title: '静态调用',
        content: '在任何地方都可以调用',
        theme: 'cyber',
    });

    // 3秒后自动关闭
    setTimeout(() => {
        GlobalModalManager.close(modalId);
    }, 3000);
}
````


组件直接使用

````typescript jsx
import Modal from '@/components/ui/cyber-modal';

function MyComponent() {
    const [isOpen, setIsOpen] = useState(false);

    return (
        <>
            <button onClick={() => setIsOpen(true)}>打开弹窗</button>
            <Modal
                isOpen={isOpen}
                onClose={() => setIsOpen(false)}
                title="直接使用"
                content="内容"
            />
        </>
    );
}
````
